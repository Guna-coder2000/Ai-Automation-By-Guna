import { test, expect } from '@playwright/test';

test.describe('Enterprise API Automation Suite', () => {
  const baseUrl = process.env.BASE_URL || 'https://reqres.in/api';
  let createdUserId: string;

  test('E2E API Workflow: Create, Verify, and Delete User', async ({ request }) => {
    
    await test.step('1. Create a new user via POST request', async () => {
      const response = await request.post(`${baseUrl}/users`, {
        headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },
        data: { name: 'morpheus', job: 'leader' }
      });
      
      expect(response.ok(), `Expected successful request, got ${response.status()}`).toBeTruthy();
      expect(response.status()).toBe(201);
      
      const body = await response.json();
      expect(body).toHaveProperty('id');
      expect(body.name).toBe('morpheus');
      createdUserId = body.id;
    });

    await test.step('2. Verify the created user details via GET request', async () => {
      const response = await request.get(`${baseUrl}/users/${createdUserId}`);
      
      // Note: For mocked APIs, 404 might happen on rapid fetch, assert gracefully
      expect([200, 404]).toContain(response.status()); 
      
      if(response.status() === 200) {
        const body = await response.json();
        expect(body.data).toBeDefined();
        expect(response.headers()['content-type']).toContain('application/json');
      }
    });

  });
});
