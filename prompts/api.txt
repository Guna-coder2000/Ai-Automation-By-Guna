import { test, expect } from '@playwright/test';
import { CommonApiActions } from '../../src/framework/CommonApiActions';

test.describe('Enterprise API Automation Suite', () => {
  const baseUrl = process.env.BASE_URL || 'https://reqres.in/api';
  let createdUserId: string;

  test('E2E API Workflow: Create, Verify, and Delete User', async ({ request }) => {
    const api = new CommonApiActions(request);
    
    await test.step('1. Create a new user via POST request', async () => {
      const responseBody = await api.post(`${baseUrl}/users`, 
        { name: 'morpheus', job: 'leader' }, 
        201, 
        { 'Content-Type': 'application/json', 'Accept': 'application/json' }
      );
      
      expect(responseBody).toHaveProperty('id');
      expect(responseBody.name).toBe('morpheus');
      createdUserId = responseBody.id;
    });

    await test.step('2. Verify the created user details via GET request', async () => {
      const responseBody = await api.get(`${baseUrl}/users/${createdUserId}`, 200);
      expect(responseBody.data).toBeDefined();
    });

  });
});
